	function YearToDate = 
			(
						CalcValue : ANYREF EXPR,
			        	dateTable : ANYREF EXPR,
			        	dateColumn : ANYREF EXPR,
			        	offsetValue : VAL
			    ) =>
			
					VAR LatestDate =
							MAX( dateColumn )  
					VAR LatestDay = DAY( LatestDate )  
			VAR PriorYearStart =
			     DATE
			     (
			          YEAR( LatestDate ) - offsetValue,
			          1,
			          1
			     ) 
			VAR PriorYearEnd =
			     DATE
			     (
			          YEAR( LatestDate ) - offsetValue,
			          MONTH( LatestDate ),
			          LatestDay
			     )  
			
			VAR Result =
				IF ( 
					ISBLANK(CalcValue), 
					BLANK(), 
						CALCULATE
						(
							CalcValue,
							ALL( dateTable ),
							dateColumn >= PriorYearStart
								&& dateColumn <= PriorYearEnd
						)
				)
			
			RETURN 
				Result
			
    annotation PackageId = FosterBIServices.YearToDate
