DEFINE
	// This function evaluates a Priior Year to Date result based on the selected measure.  
	FUNCTION PriorYearToDate = (
			CalcValue : ANYREF EXPR,
        	dateTable : ANYREF EXPR,
        	dateColumn : ANYREF EXPR,
        	offsetValue : VAL
    ) =>

		VAR LatestDate =
				MAX( dateColumn )  
		VAR LatestDay = DAY( LatestDate )  
VAR PriorYearStart =
     DATE
     (
          YEAR( LatestDate ) - 1,
          1,
          1
     ) 
VAR PriorYearEnd =
     DATE
     (
          YEAR( LatestDate ) - 1,
          MONTH( LatestDate ),
          LatestDay
     )  

VAR Result =
     CALCULATE
     (
          CalcValue,
		  ALL( dateTable ),
          dateColumn >= PriorYearStart
               && dateColumn <= PriorYearEnd
			
           
     )
RETURN 
	Result
